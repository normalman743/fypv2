# 文件架构重构建议 - 可持续扩展设计

## 1. 核心问题分析

### 当前架构问题
- ❌ `files` 和 `global_files` 表重复
- ❌ 文件共享机制缺失
- ❌ 权限控制不够灵活
- ❌ 扩展性受限

### 目标
- ✅ 统一文件管理
- ✅ 灵活的共享机制
- ✅ 细粒度权限控制
- ✅ 支持未来扩展

## 2. 重构方案

### 2.1 统一文件表设计

```python
class File(Base):
    __tablename__ = "files"

    id = Column(Integer, primary_key=True, index=True)
    physical_file_id = Column(Integer, ForeignKey("physical_files.id"), nullable=False)
    
    # 基本信息
    original_name = Column(String(255), nullable=False)
    file_type = Column(String(50), nullable=False)
    description = Column(Text, nullable=True)
    tags = Column(JSON, nullable=True)
    
    # 作用域和归属 (核心改进)
    scope = Column(String(20), nullable=False, default='course', index=True)
    # scope 取值: 'course', 'global', 'personal', 'shared'
    
    owner_id = Column(Integer, ForeignKey("users.id"), nullable=False, index=True)
    course_id = Column(Integer, ForeignKey("courses.id"), nullable=True, index=True)
    folder_id = Column(Integer, ForeignKey("folders.id"), nullable=True, index=True)
    
    # 可见性控制
    visibility = Column(String(20), default='private', index=True)
    # visibility 取值: 'private', 'course', 'public', 'shared'
    
    # 权限和共享
    is_shareable = Column(Boolean, default=True)
    share_settings = Column(JSON, nullable=True)  # 存储复杂共享配置
    
    # RAG和处理
    is_processed = Column(Boolean, default=False)
    processing_status = Column(String(20), default="pending")
    processing_error = Column(Text, nullable=True)
    processed_at = Column(DateTime, nullable=True)
    chunk_count = Column(Integer, default=0)
    content_preview = Column(Text, nullable=True)
    
    # 元数据
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())
    
    # 索引优化
    __table_args__ = (
        Index('idx_scope_visibility', scope, visibility),
        Index('idx_owner_course', owner_id, course_id),
    )
```

class FileShare(Base):
    __tablename__ = "file_shares"
    
    id = Column(Integer, primary_key=True)
    file_id = Column(Integer, ForeignKey("files.id"), nullable=False, index=True)
    
    # 共享对象 (灵活设计)
    shared_with_type = Column(String(20), nullable=False)  
    # 取值: 'user', 'course', 'group', 'role', 'public'
    shared_with_id = Column(Integer, nullable=True)  # 根据type决定是否需要
    
    # 权限级别
    permission_level = Column(String(20), default='read')
    # 取值: 'read', 'comment', 'edit', 'manage'
    
    # 共享配置
    can_reshare = Column(Boolean, default=False)
    download_allowed = Column(Boolean, default=True)
    expires_at = Column(DateTime, nullable=True)
    
    # 追踪信息
    shared_by = Column(Integer, ForeignKey("users.id"), nullable=False)
    created_at = Column(DateTime, server_default=func.now())
    last_accessed = Column(DateTime, nullable=True)
    access_count = Column(Integer, default=0)
    
    __table_args__ = (
        Index('idx_share_target', shared_with_type, shared_with_id),
        Index('idx_file_permissions', file_id, permission_level),
    )

### 4.2 RAG知识库权限
```python
def get_accessible_chunks_for_rag(user_id: int, course_id: int = None):
    """获取用户可访问的文档chunks用于RAG检索"""
    
    accessible_files = session.query(File.id).filter(
        or_(
            # 用户拥有的文件
            File.owner_id == user_id,
            
            # 课程文件 (如果指定了course_id)
            and_(File.course_id == course_id, File.scope == 'course') if course_id else False,
            
            # 共享给用户的文件
            and_(
                File.id.in_(
                    session.query(FileShare.file_id).filter(
                        FileShare.shared_with_type == 'user',
                        FileShare.shared_with_id == user_id
                    )
                )
            ),
            
            # 全局公开文件
            and_(File.scope == 'global', File.visibility == 'public')
        )
    )
    
    return session.query(DocumentChunk).filter(
        DocumentChunk.file_id.in_(accessible_files)
    )
```

### 5.1 数据迁移策略
```sql
-- 步骤1: 创建新的统一文件表
CREATE TABLE files_new LIKE files;
ALTER TABLE files_new ADD COLUMN scope VARCHAR(20) DEFAULT 'course';
ALTER TABLE files_new ADD COLUMN visibility VARCHAR(20) DEFAULT 'private';

-- 步骤2: 迁移现有数据
INSERT INTO files_new (
    physical_file_id, original_name, file_type, scope, 
    course_id, folder_id, owner_id, visibility, ...
)
SELECT 
    physical_file_id, original_name, file_type, 'course',
    course_id, folder_id, user_id, 'course', ...
FROM files;

INSERT INTO files_new (
    file_hash, original_name, file_type, scope,
    NULL, NULL, created_by, visibility, ...
)
SELECT 
    file_hash, original_name, 'pdf', 'global',
    NULL, NULL, created_by, 
    CASE WHEN is_public THEN 'public' ELSE 'private' END, ...
FROM global_files;

-- 步骤3: 切换表名
RENAME TABLE files TO files_old;
RENAME TABLE files_new TO files;
```

## 7. 总结

这个重构方案的核心优势:

1. **统一管理**: 一个表管理所有文件，减少复杂性
2. **灵活共享**: 支持多种共享场景和权限级别
3. **可扩展性**: 为未来功能预留扩展空间
4. **性能优化**: 合理的索引和查询优化
5. **安全性**: 完善的权限控制和审计

建议**分阶段实施**，先完成核心的表合并，再逐步添加共享功能，确保系统稳定性。
